 11 /** @class
 12 
 13 An Array Controller provides a way to project the contents of an array
 14 out to a view.  You can use this object any place you might use an
 15 array.  Changes to the array will not propogate to the content array
 16 until you call commitChanges().
 17 
 18 @extends SC.Controller
 19 @extends SC.Array
 20 @extends SC.SelectionSupport
 21 @since SproutCore 1.0
 22 
 23 */
 24 SC.ArrayController = SC.Controller.extend(SC.Array, SC.SelectionSupport,
 25 /** @scope SC.ArrayController.prototype */
 26 {
 27   /**
 28     If YES the will return controllers for content objects.
 29     
 30     If you want to use an array controller to edit an array contents directly
 31     but you do not want to wrap the values of the array in controller objects
 32     then you should set this property to NO.
 33     
 34     @field
 35     @type {Boolean}
 36   */
 37   useControllersForContent: NO,
 38   
 39   /**
 40     Provides compatibility with CollectionControllers.
 41     @field
 42     @type {SC.ArrayController}
 43   */
 44   arrangedObjects: function() { return this; }.property('content'),
 45 
 46   /**
 47     The array content that is being managed by the controller.
 48     @field
 49     @type {Array}
 50   */
 51   content: null,
 52   contentBindingDefault: SC.Binding.Multiple,
 53   
 54   /**
 55     Set to true if you want objects removed from the array to also be
 56     deleted.  This is a convenient way to manage lists of items owned
 57     by a parent record object.
 58     
 59     @field
 60     @type {Boolean}
 61   */
 62   destroyOnRemoval: NO,
 63     
 64   /**
 65     Watches changes to the content property updates the contentClone.
 66     @private
 67     @observes content
 68   */
 69   _contentObserver: function() {
 70     var content = this.get('content') ;
 71     if (SC.isEqual(content, this._content)) return ; // nothing to do
 72 
 73     if (!this._boundContentPropertyObserver) {
 74       this._boundContentPropertyObserver = this._contentPropertyObserver.bind(this) ;
 75     }
 76     var func = this._boundContentPropertyObserver ;
 77 
 78     // remove old observer, add new observer, and trigger content property change
 79     if (this._content && this._content.removeObserver) this._content.removeObserver('[]', func) ;
 80     if (content && content.addObserver) content.addObserver('[]', func) ;
 81     this._content = content; //cache
 82     this._contentPropertyRevision = null ;
 83     
 84     var rev = (content) ? content.propertyRevision : -1 ;
 85     this._contentPropertyObserver(this, '[]', content, rev) ; 
 86   }.observes('content'),
 87 
 88   _contentPropertyObserver: function(target, key, value, rev) {  
 89       
 90     if (!this._updatingContent && (!rev || (rev != this._contentPropertyRevision))) {
 91       this._contentPropertyRevision = rev ;
 92 
 93       this._updatingContent = true ;
 94 
 95       this.beginPropertyChanges();
 96       this.contentCloneReset();
 97       this.arrayContentDidChange() ;
 98       this.notifyPropertyChange('length') ;
 99       this.updateSelectionAfterContentChange();
100       this.endPropertyChanges() ;
101 
102       this._updatingContent = false ;
103 
104     }
105   },
106   
107   /**
108     The array content that (when committed) will be merged back into the 
109     content property. All array methods will take place on this object.
110 
111     @field
112     @type {SC.Array}
113   */
114   contentClone: null,
115 
116   /**
117     Clones the content property into the contentClone property.
118     @private
119   */
120   contentCloneReset: function() {
121     this._changelog = [];
122     this.set('contentClone', null);
123   },
124 
125   /**
126    SC.Array interface implimentation.
127    
128    @param idx {Number} Starting index in the array to replace.  If idx >= 
129      length, then append to the end of the array.
130    
131    @param amt {Number} Number of elements that should be removed from the 
132      array, starting at *idx*.
133    
134    @param objects {Array} An array of zero or more objects that should be 
135      inserted into the array at *idx* 
136   */
137   replace: function(idx, amt, objects) {
138 
139     var content = this.get('content') ;
140 
141     // in case the passed objects are controllers, convert to source objects.
142     var copyIdx = objects.length ;
143     var sourceObjects = [] ;
144     while(--copyIdx >= 0) sourceObjects[copyIdx] = this._sourceObjectFor(objects[copyIdx]) ;
145     
146     // create clone of content array if needed
147     var contentClone = this.get('contentClone') ;
148     if (!contentClone) {
149       this.set('contentClone', contentClone = content.clone()) ;
150     }
151 
152     // now, record the removed objects.  This may be used later.
153     if (this.get('destroyOnRemoval')) {
154       if (!this._deletions) this._deletions = [] ;
155       for (var i=0; i < amt; i++) {
156         this._deletions.push(content.objectAt(idx + i));
157       }
158     }
159 
160     // and record additions
161     if (!this._changelog) this._changelog = []; 
162     this._changelog.push({ idx: idx, amt: amt, objects: sourceObjects });
163     
164     // then actually perform the edit on the contentClone
165     contentClone.replace(idx, amt, sourceObjects);
166     
167     this.editorDidChange() ;
168     this.arrayContentDidChange();
169     this.updateSelectionAfterContentChange();
170     
171     return this;
172   },
173   /**
174   * SC.Array interface implimentation.
175   * 
176   * @param {Number} idx
177   *   The index of the item to return.  If idx exceeds the current length, 
178   *   return null.
179   */
180   objectAt: function(idx) {
181     var obj = this._getSourceContent() ;
182     obj = (obj && obj.objectAt) ? obj.objectAt(idx) : null;
183     return this._objectControllerFor(obj) ;
184   },
185   /**
186   * SC.Array interface implimentation.
187   * @field
188   * @type {integer}
189   */
190   length: function( key, value ) {
191     var ret = this._getSourceContent() ;
192     return (ret && ret.get) ? (ret.get('length') || 0) : 0 ;
193   }.property(),
194 
195   /**
196     Returns the index in the array of the specified object.
197     
198     This can handle both controller wrapper objects and source content objects.
199   */
200   indexOf: function( obj ) {
201     return this._getSourceContent().indexOf(this._sourceObjectFor(obj)) ;
202   },
203   
204   _getSourceContent: function() {
205     return this.get('contentClone') || this.get('content') || [];
206   },
207   
208   /** 
209   * @private
210   */
211   performCommitChanges: function()
212   {
213     var content = this.get('content');
214     var ret     = true;
215     
216     // cannot commit changes to null content.  Return an error.
217     if (!content) {
218       return $error("No Content");
219     }
220     
221     if (content.beginPropertyChanges) content.beginPropertyChanges();
222 
223 
224     // apply all the changes made to the clone
225     var changelog = this._changelog || [] ;
226     for(var idx=0;idx<changelog.length;idx++) {
227       var change = changelog[idx];
228       content.replace(change.idx, change.amt, change.objects) ;
229     }
230     
231     // done, flush the changelog
232     this._changelog = [] ;
233 
234     // finally, destroy any removed objects if necessary.  Make 
235     // sure the objects have not been re-added before doing this.
236     if (this.get('destroyOnRemoval') && this._deletions && this._deletions.length>0) {
237       var idx = this._deletions.length;
238       while(--idx >= 0) {
239         var obj = this._deletions[idx] ;
240         if (obj && obj.destroy && (content.indexOf(obj) < 0)) {
241           obj.destroy() ; 
242         }
243       }
244       this._deletions = [] ; // clear array
245     }
246     
247      // finish commiting changes.
248     if (content.endPropertyChanges) content.endPropertyChanges();
249     if (content.commitChanges) ret = content.commitChanges();
250     
251     if ($ok(ret)) {
252       this.contentCloneReset();
253       this.editorDidClearChanges();
254     }
255     
256     return ret;
257   },
258   /** 
259   * @private
260   */
261   performDiscardChanges: function()
262   {
263     this.contentCloneReset();
264     this.editorDidClearChanges();
265     return true;
266   },
267   
268   /** @private
269     Returns the object controller for a source value.
270   */
271   _objectControllerFor: function(obj) {
272     if (!this.useControllersForContent) return obj;
273     
274     var controllers = this._objControllers = this._objControllers || {} ;
275     var guid = SC.getGUID(obj) ;
276     var ret = controllers[guid] ;
277     if (!ret) {
278       ret = controllers[guid] = this.controllerForValue(obj) ;
279       if (ret) ret.__isArrayController = true ;
280     }
281     return ret ;
282   },
283   
284   /** @private
285     Returns the source object for the passed value.  If the passed value is a 
286     controller, this will map back to the sourceo object.  Otherwise the object itself
287     will be returned.
288   */
289   _sourceObjectFor: function(obj) {
290     return (obj && obj.kindOf && obj.kindOf(SC.Controller)) ? obj.get('content') : obj ;
291   },
292   
293   init: function() {
294     sc_super() ;
295     if (this.get('content')) this._contentObserver() ;
296   }
297 
298 });
