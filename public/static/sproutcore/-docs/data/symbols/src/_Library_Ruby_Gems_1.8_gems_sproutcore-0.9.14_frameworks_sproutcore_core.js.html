118   // Call this method during setup of your app to queue up methods to be 
119   // called once the entire document has finished loading.  If you call this
120   // method once the document has already loaded, then the function will be
121   // called immediately.
122   callOnLoad: function(func) { 
123     if (SC._onloadQueueFlushed) func.call(document);
124     var queue = SC._onloadQueue || [] ;
125     queue.push(func) ; SC._onloadQueue = queue ;
126 	queue = null;
127   },
128 
129   // To flush the callOnLoad queue, you need to set window.onload=SC.didLoad
130   didLoad: function() { 
131     SC.app = SC.Application.create();
132     SC.app.run();
133     
134     // set the current language
135     var b = $tag('body');
136     Element.addClassName(b, String.currentLanguage().toLowerCase()) ;
137 
138     // call the onloadQueue.
139     var queue ;
140     SC.runLoop.beginRunLoop() ;
141     if (window.callOnLoad) {
142       if (window.callOnLoad instanceof Array) {
143         queue = window.callOnLoad ;
144       } else if (window.callOnLoad instanceof Function) {
145         queue = [window.callOnLoad] ;
146       }
147     } else queue = [] ;
148     queue = queue.concat(SC._onloadQueue) ;
149     var func = null ;
150     while(func = queue.shift()) func.call(document) ;
151     SC._onloadQueueFlushed = true ;
152         
153     // start the app; call main.
154     if (window.main && (main instanceof Function)) main() ; // start app.
155     
156     // finally handle any routes if any.
157     if (typeof Routes != 'undefined') {
158       Routes.doRoutes() ; // old style.
159     } else if (typeof SC.Routes != 'undefined') {
160       SC.Routes.ping() ; // handle routes, if modules is installed.
161     }
162     
163     SC.runLoop.endRunLoop();
164 		//remove possible IE7 leak
165 		b = null;
166 		queue = null;
167 		func = null;
168   },

170   // this will take a URL of any type and convert it to a fully qualified URL.
171   normalizeURL: function(url) {
172     if (url.slice(0,1) == '/') {
173       url = window.location.protocol + '//' + window.location.host + url ;
174     } else if ((url.slice(0,5) == 'http:') || (url.slice(0,6) == 'https:')) {
175       // no change
176     } else {
177       url = window.location.href + '/' + url ;
178     }
179     return url ;
180   },

182   // use this instead of typeOf() to get the type of item.  The return values
183   // are: 'string', 'number', 'function', 'class', 'object', 'hash', 'null', 
184   // 'undefined', 'boolean'.  
185   // 'object' will be returned for any items inheriting from SC.Object. 'hash' 
186   // is any other type of object.
187   typeOf: function(item) {
188     if (item === undefined) return T_UNDEFINED ;
189     if (item === null) return T_NULL ; 
190     var ret = typeof(item) ;
191     if (ret == "object") {
192       if (item instanceof Array) {
193         ret = T_ARRAY ;
194       } else if (item instanceof Function) {
195         ret = (item.isClass) ? T_CLASS : T_FUNCTION ;
196       } else if (item instanceof SC.Error) {
197         ret = T_ERROR ;        
198       } else if (item.isObject === true) {
199         ret = T_OBJECT ;
200       } else ret = T_HASH ;
201     } else if (ret === T_FUNCTION) ret = (item.isClass) ? T_CLASS : T_FUNCTION;
202     return ret ;
203   },

205   // this will compare two values to see if they are equal.  If you have two 
206   // values of unknown type, this is faster across all browsers than ===.
207   isEqual: function(a,b) {
208     if (a === null) {
209       return b === null ;
210     } else if (a === undefined) {
211       return b === undefined ;
212     } else if (typeof(a) == typeof(b)) {
213       return a == b ;
214     }

367   // this will serialize a general JSON object into a URI.
368   serialize: function(obj) {
369     var ret = [] ;
370     for(var key in obj) {
371       var value = obj[key] ;
372       if (typeof value == 'number') { value = '' + value ; }
373       if (!(typeof value == 'string')) { value = value.join(','); }
374       ret.push(encodeURIComponent(key) + "=" + encodeURIComponent(value)) ;
375     }
376     return ret.join('&') ;
377   }

382 // This will add or remove the class name based on the flag, allowing you to
383 // treat it like a bool setting.  Simplifies the common case where you need
384 // to make a class name match a bool.
385 Element.setClassName = function(element,className,flag) {
386   if(SC.isIE())
387   {
388     if (flag) { 
389       Element.addClassName(element,className); 
390     } else {
391       Element.removeClassName(element,className) ;
392     }
393   } 
394   else
395   {
396     if (flag) { 
397       element.addClassName(className); 
398     } else {
399       element.removeClassName(className) ;
400     }
401   }
