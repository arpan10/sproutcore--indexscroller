  6 // These are basic enhancements to the string class used throughout 
  7 // SproutCore.
  8 // capitalize a string.
  9 Object.extend(String.prototype,{
 10 
 11   // Capitalize a string.
 12   //
 13   // mode: optional.  'each' - capitalize each word. the default.
 14   //                  'first' - capitalize the first word only.
 15   
 16   capitalize: function(mode) {
 17     var words = (mode == 'first') ? this : this.split(' ') ;
 18     words = words.map(function(word) {
 19       if (word.length == 0) return word ;
 20       return word.charAt(0).toUpperCase() + word.substring(1) ;
 21     }) ;
 22     return words.join(' ') ;
 23   },
 24 
 25   // Interpolate string. looks for %@ or %@1; to control the order of params.
 26   format: function() {
 27     var args = $A(arguments) ;
 28     
 29     // first, replace any ORDERED replacements.
 30     var str = this.gsub(/%@([0-9]+)/, function(m) {
 31       return (args[parseInt(m[1],0)-1] || '').toString(); 
 32     }) ;
 33 
 34     // now, replace any remaining %@ items.  Use this indexOf() method b/c
 35     // it is faster than split().
 36     var ret = [] ;
 37     var idx = -1 ;
 38     var loc = 0 ;
 39     while((idx = str.indexOf("%@",loc)) >= 0) {
 40      // slice off initial part of string and push into ret. update loc.
 41      ret.push(str.slice(loc,idx)) ;
 42      loc = idx + 2 ; // 2 to skip '%@'.
 43      
 44      // add in replacement.
 45      var value = args.shift() ;
 46      if (value && value.toString) value = value.toString() ;
 47      ret.push(value) ;
 48     }
 49     
 50     // include any remaining bits of the string.
 51     if (loc < str.length) {
 52       ret.push(str.slice(loc,str.length)) ;
 53     }
 54     
 55     // join return value.
 56     return (ret.length > 1) ? ret.join('') : ret[0] ;
 57   },
 58 
 59   // localize a string.  Also interpolates any items you pass just like
 60   // format().
 61   loc: function() {
 62     // NB: This could be implemented as a wrapper to locWithDefault() but
 63     // it would add some overhead to deal with the arguments and adds stack
 64     // frames, so we are keeping the implementation separate.
 65     
 66     var kit = String[String.currentLanguage()];
 67     var str = kit[this] ;
 68     if (!str) str = String.English[this] || this ;
 69     return str.format.apply(str,arguments) ;
 70   },
 71   
 72   // this works just like loc except it will return the first argument
 73   // as a default if the matching value is not found.
 74   locWithDefault: function(def) {
 75     var kit = String[String.currentLanguage()];
 76     var str = kit[this] ;
 77     if (!str) str = String.English[this] || def ;
 78     var args = $A(arguments) ;
 79     args.shift() ; // escape def.
 80     return str.format.apply(str, args) ;
 81   },
 82   
 83   classify: function() {
 84     return this.camelize().capitalize() ;
 85   },
 86   
 87   decamelize: function() { 
 88     return this.replace(/([a-z])([A-Z])/g,'$1_$2').toLowerCase();
 89   },
 90 
 91   dasherize: function() {
 92     return this.decamelize().replace(/[ _]/g,'-') ;  
 93   },
 94   
 95   humanize: function() {
 96     return this.decamelize().replace(/[-_]/g,' ') ;
 97   },
 98   
 99   toHref: function() {
100     if (this.match(/.+@.+\...+/)) {
101       return 'mailto:' + this;
102     } else if (this.indexOf('http://') != 0 && this.indexOf('https://') !=0 && this.match(/[^.]+\.[^.]+/)) {
103       return 'http://' + this;
104     } else {
105       return this;
106     }
107   },
108   
109   trim: function ()
110   {
111     return this.replace(/^\s+|\s+$/g,"");
112   },
113   
114   strip: function()
115   {
116     return this.trim();
117   }
118   
119 }) ;
120 
121 // Shorter alias of the format function.
122 String.prototype.fmt = String.prototype.format ;

124 // Add strings for various languages to this collection.  String.loc()
125 // method will try to localize the string passed using the current language.
126 // if the language is not available, it will use English.
127 Object.extend(String,
128 /** @scope String @static */ {
129 
130   /**
131     The current browser language as a two letter code.
132   */
133   browserLanguage: ((navigator.language || navigator.browserLanguage).split('-', 1)[0]),
134   
135   /**
136     If YES, localization will favor the detected language instead of the
137     preferred one.
138   */
139   useAutodetectedLanguage: NO,
140   
141   /**
142     This property is set by the build tools to the current build language.
143   */
144   preferredLanguage: null,
145   
146   /**
147     Returns the hash key to use for loc strings.  The default implementation
148     will autodetect the browser language and look for a loc string to 
149     match.  If it can't find one then it will introspect to find loc strings
150     that are defined and use those instead.
151   */
152   currentLanguage: function () {
153     var ret = (this.useAutodetectedLanguage) ? (this.browserLanguage || this.preferredLanguage || 'en') : (this.preferredLanguage || this.browserLanguage || 'en') ;
154 
155     // then try a couple of normalized forms...
156     if (!this[ret]) ret = this.normalizedLanguage(ret);
157     return ret ;
158   },
159   
160   /**
161     Returns a normalized language string for the two letter country code.
162   */
163   normalizedLanguage: function(ret) {
164     switch(ret) {
165       case 'fr':
166         ret = 'French'; 
167         break ;
168       case 'de':
169         ret = 'German'; 
170         break ;
171       case 'ja':
172       case 'jp':
173         ret = 'Japanese'; 
174         break ;
175       case 'en':
176         ret = 'English' ;
177         break ;
178       
179       case 'es':
180         ret = 'Spanish' ;
181         break;
182         
183       default:
184         ret = "English";
185         break ;
186     }
187     return ret;
188   },
189   
190   /**
191     Adds loc strings for the named language.  This method takes care of 
192     creating the localized string hash if it does not already exist.
193     The language can be one of the following or any two-letter country code.
194     
195     English, French, German, Japanese, Spanish
196     
197     @param language {String} the language code
198     @param strings {Hash} hash of loc strings.
199     @returns {this}
200   */
201   addStringsFor: function(language, strings) {    
202     // convert language to a normalized name...
203     language = String.normalizedLanguage(language) ;
204     if (!String[language]) String[language] = {} ;
205     Object.extend(String[language], strings || {}); 
206     return this;
207   }
208 
209 });
210 
211 String.English  = String.English  || {};
212 String.French   = String.French   || {};
213 String.German   = String.German   || {};
214 String.Japanese = String.Japanese || {};
