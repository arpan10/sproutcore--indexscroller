  6 // These are helpful utility functions for calculating range and rect values
  7 
  8 
  9 Object.extend(SC, 
 10 /** @scope SC */
 11 {
 12 
 13   /** Return the left edge of the frame */
 14   minX: function(frame) { 
 15     return frame.x; 
 16   },
 17   
 18   /** Return the right edge of the frame. */
 19   maxX: function(frame) { 
 20     return frame.x + frame.width; 
 21   },
 22   
 23   /** Return the midpoint of the frame. */
 24   midX: function(frame) {
 25     return frame.x + (frame.width / 2) ;
 26   },
 27   
 28   /** Return the top edge of the frame */
 29   minY: function(frame) {
 30     return frame.y ;
 31   },
 32   
 33   /** Return the bottom edge of the frame */
 34   maxY: function(frame) {
 35     return frame.y + frame.height ;
 36   },
 37   
 38   /** Return the midpoint of the frame */
 39   midY: function(frame) {
 40     return frame.y + (frame.height / 2) ;
 41   },
 42   
 43   /** Returns the point that will center the frame X within the passed frame. */
 44   centerX: function(innerFrame, outerFrame) {
 45     return (outerFrame.width - innerFrame.width) / 2 ;
 46   },
 47   
 48   /** Return the point that will center the frame Y within the passed frame. */
 49   centerY: function(innerFrame, outerFrame) {
 50     return (outerFrame.width - innerFrame.width) /2  ;
 51   },
 52   
 53   /** Check if the given point is inside the rect. */
 54   pointInRect: function(point, f) {
 55     return  (point.x >= SC.minX(f)) &&
 56             (point.y >= SC.minY(f)) &&
 57             (point.x <= SC.maxX(f)) && 
 58             (point.y <= SC.maxY(f)) ;
 59   },
 60   
 61   /** Return true if the two frames match.
 62   
 63     @param r1 {Rect} the first rect
 64     @param r2 {Rect} the second rect
 65     @param delta {Float} an optional delta that allows for rects that do not match exactly. Defaults to 0.1
 66     @returns {Boolean} true if rects match
 67    */
 68   rectsEqual: function(r1, r2, delta) {
 69     if (!r1 || !r2) return (r1 == r2) ;
 70     
 71     if (delta == null) delta = 0.1;
 72     if (Math.abs(r1.y - r2.y) > delta) return false ; 
 73     if (Math.abs(r1.x - r2.x) > delta) return false ; 
 74     if (Math.abs(r1.width - r2.width) > delta) return false ; 
 75     if (Math.abs(r1.height - r2.height) > delta) return false ; 
 76     return true ;
 77   },
 78   
 79   /** Returns the insersection between two rectangles. 
 80   
 81     @param r1 {Rect} The first rect
 82     @param r2 {Rect} the second rect
 83     @returns {Rect} the intersection rect.  width || height will be 0 if they do not interset.
 84   */
 85   intersectRects: function(r1, r2) {
 86     // find all four edges
 87     var ret = {
 88       x: Math.max(SC.minX(r1), SC.minX(r2)),
 89       y: Math.max(SC.minY(r1), SC.minY(r2)),
 90       width: Math.min(SC.maxX(r1), SC.maxX(r2)),
 91       height: Math.min(SC.maxY(r1), SC.maxY(r2))
 92     } ;
 93     
 94     // convert edges to w/h
 95     ret.width = Math.max(0, ret.width - ret.x) ;
 96     ret.height = Math.max(0, ret.height - ret.y) ;
 97     return ret ;
 98   },
 99   
100   /** Returns the union between two rectangles
101   
102     @param r1 {Rect} The first rect
103     @param r2 {Rect} The second rect
104     @returns {Rect} The union rect.
105   */
106   unionRects: function(r1, r2) {
107     // find all four edges
108     var ret = {
109       x: Math.min(SC.minX(r1), SC.minX(r2)),
110       y: Math.min(SC.minY(r1), SC.minY(r2)),
111       width: Math.max(SC.maxX(r1), SC.maxX(r2)),
112       height: Math.max(SC.maxY(r1), SC.maxX(r2))
113     } ;
114     
115     // convert edges to w/h
116     ret.width = Math.max(0, ret.width - ret.x) ;
117     ret.height = Math.max(0, ret.height - ret.y) ;
118     return ret ;
119   },
120   
121   /** Duplicates the passed rect.  
122   
123     This is faster than Object.clone(). 
124     
125     @param r {Rect} The rect to clone.
126     @returns {Rect} The cloned rect
127   */
128   cloneRect: function(r) {
129     return { x: r.x, y: r.y, width: r.width, height: r.height } ;
130   },
131   
132   
133   /** Finds the absolute viewportOffset for a given element.
134     This method is more accurate than the version provided by prototype.
135     
136     If you pass NULL to this method, it will return a { x:0, y:0 }
137     @param el The DOM element
138     @returns {Point} A hash with x,y offsets.
139   */
140   viewportOffset: function(el) {
141     var valueL = 0 ; var valueT = 0;
142 
143     // add up all the offsets for the element.
144     var element = el ;
145     while (element) {
146       valueT += (element.offsetTop  || 0) + (element.clientTop  || 0);
147       valueL += (element.offsetLeft || 0) + (element.clientLeft || 0);
148 
149       // bizarely for FireFox if your offsetParent has a border, then it can 
150       // impact the offset. 
151       if (SC.Platform.Firefox) {
152         var overflow = Element.getStyle(element, 'overflow') ;
153         if (overflow !== 'visible') {
154           var left = parseInt(Element.getStyle(element, 'borderLeftWidth'),0) || 0 ;
155           var top = parseInt(Element.getStyle(element, 'borderTopWidth'),0) || 0 ;
156           if (el !== element) {
157             left *= 2; top *= 2 ;
158           }
159           valueL += left; valueT += top ;
160         }
161       }
162 
163       // Safari fix
164       if (element.offsetParent == document.body &&
165         Element.getStyle(element, 'position') == 'absolute') break;
166 
167       element = element.offsetParent ;
168 
169     }
170 
171     element = el;
172     while (element) {
173       if (!Prototype.Browser.Opera || element.tagName == 'BODY') {
174         valueT -= element.scrollTop  || 0;
175         valueL -= element.scrollLeft || 0;
176       }
177       
178       element = element.parentNode ;
179     }
180 
181     return { x: valueL, y: valueT } ;
182   },
183   
184   /** A Point at {0,0} */
185   ZERO_POINT: { x: 0, y: 0 },
186   
187   /** A zero length range at zero. */
188   ZERO_RANGE: { start: 0, length: 0 },
189 
190   RANGE_NOT_FOUND: { start: 0, length: -1 },
191   
192   /** Returns true if the passed index is in the specified range */
193   valueInRange: function(value, range) {
194     return (range > 0) && (value >= range.start) && (value < (range.start + range.length));  
195   },
196   
197   /** Returns first value of the range. */
198   minRange: function(range) { return range.start; },
199   
200   /** Returns the first value outside of the range. */
201   maxRange: function(range) { return (range.length < 0) ? -1 : (range.start + range.length); },
202   
203   /** Returns the union of two ranges.  If one range is null, the other
204    range will be returned.  */
205   unionRanges: function(r1, r2) { 
206     if ((r1 == null) || (r1.length < 0)) return r2 ;
207     if ((r2 == null) || (r2.length < 0)) return r1 ;
208     
209     var min = Math.min(r1.start, r2.start) ;
210     var max = Math.max(SC.maxRange(r1), SC.maxRange(r2)) ;
211     return { start: min, length: max - min } ;
212   },
213   
214   /** Returns the intersection of the two ranges or SC.RANGE_NOT_FOUND */
215   intersectRanges: function(r1, r2) {
216     if ((r1 == null) || (r2 == null)) return SC.RANGE_NOT_FOUND ;
217     if ((r1.length < 0) || (r2.length < 0)) return SC.RANGE_NOT_FOUND;
218     var min = Math.max(SC.minRange(r1), SC.minRange(r2)) ;
219     var max = Math.min(SC.maxRange(r1), SC.maxRange(r2)) ;
220     if (max < min) return SC.RANGE_NOT_FOUND ;
221     return { start: min, length: max-min };
222   },
223   
224   /** Returns a clone of the range. */
225   cloneRange: function(r) { 
226     return { start: r.start, length: r.length }; 
227   },
228   
229   /** Returns true if the two passed ranges are equal.  A null value is
230     treated like RANGE_NOT_FOUND.
231   */
232   rangesEqual: function(r1, r2) {
233     if (r1===r2) return true ;
234     if (r1 == null) return r2.length < 0 ;
235     if (r2 == null) return r1.length < 0 ;
236     return (r1.start == r2.start) && (r1.length == r2.length) ;
237   }
238   
239 }) ;
